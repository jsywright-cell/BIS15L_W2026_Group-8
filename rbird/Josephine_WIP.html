Top five bird observations between 2019 and 2021

top_five_total <- total_data %>%
  filter(year_y %in% c(2019, 2020, 2021)) %>% 
  group_by(common_name) %>%
  summarise(count = n()) %>%
  arrange(desc(count)) %>%
  slice_head(n = 5)
top_five_total
## # A tibble: 5 × 2
##   common_name           count
##   <chr>                 <int>
## 1 Black Phoebe          17705
## 2 House Finch           14295
## 3 Mourning Dove         13683
## 4 California Scrub-Jay  13598
## 5 White-crowned Sparrow 12009
top_five_total %>% 
ggplot(aes(x = common_name, y = count, fill = common_name)) +
  geom_bar(stat = "identity") +
  coord_flip()+
  scale_fill_brewer(palette = "Dark2")+
  labs(title = "Top Five Birds with Most Observations",
       x = "Common Name",
       y = "Count of Observations",
       fill="Common Name") +
  theme(plot.title = element_text(size = 15),
        axis.title.x = element_text(size = 14),
        axis.title.y = element_text(size = 14),
        legend.position = "none")
